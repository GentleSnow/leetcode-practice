```Python
class Solution:
    def trap(self, height: List[int]) -> int:
        ans = 0
        left = 0
        right = 0

        for i in range(1, len(height) - 1):
            left = max(left, height[i - 1])
            right = 0
            for j in height[i:]:
                right = max(right, j)
            mid = height[i]
            if left > mid and right > mid:
                ans += min(left, right) - mid

        return ans
```


### Solution 2: Dynamic Programming

Runtime: 44 ms

```Python
class Solution:
    def trap(self, height: List[int]) -> int:
        ans = 0
        l = len(height)
        leftMax = [0] * l
        rightMax = [0] * l

        for i in range(1, l):
            leftMax[i] = max(leftMax[i - 1], height[i - 1])
        for j in range(1, l):
            rightMax[l - j - 1] = max(rightMax[l - j], height[l - j])
        for a in range(l):
            if leftMax[a] > height[a] and rightMax[a] > height[a]:
                ans += min(leftMax[a], rightMax[a]) - height[a]

        return ans
```


### Solution 3: Stacks

Runtime: 40 ms

```Python
class Solution:
    def trap(self, height: List[int]) -> int:
        ans = 0
        st = []

        for i in range(len(height)):
            while len(st) > 0 and height[i] > height[st[len(st)-1]]:
                top = st[len(st)-1]
                st.pop()
                if len(st) == 0:
                    break
                distance = i - st[len(st)-1] - 1
                boundedHeight = min(height[i], height[st[len(st)-1]]) - height[top]

                ans += distance * boundedHeight
            st.append(i)

        return ans
```


### Solution 4: 2 pointers

Runtime: 40ms

```Python
class Solution:
    def trap(self, height: List[int]) -> int:
        ans = 0
        left = 0
        right = len(height) - 1
        leftMax = 0
        rightMax = 0

        while left < right:
            if height[left] < height[right]:
                if height[left] >= leftMax:
                    leftMax = height[left]
                else:
                    ans += leftMax - height[left]
                left += 1
            else:
                if height[right] >= rightMax:
                    rightMax = height[right]
                else:
                    ans += rightMax - height[right]
                right -= 1

        return ans
```
